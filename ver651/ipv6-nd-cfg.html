Schema Definitions

ipv6-neighbor_schema:

IPv6 neighbor or neighbor discovery configuration

neighbors: object

IPv6 neighbors

scavenge-timeout: integer

Set lifetime for stale neighbor

range:"1..43200" units:"second"
Example
{
  "neighbors": {
    "neighbor": [
      {
        "neighbor-address": "string",
        "mac-address": "string",
        "interface-name": "string",
        "zone": "string",
        "encapsulation": "string"
      }
    ]
  },
  "scavenge-timeout": "integer"
}

neighbors_schema:

IPv6 neighbors

neighbor: object[]

IPv6 neighbor configuration

Example
{
  "neighbor": [
    {
      "neighbor-address": "string",
      "mac-address": "string",
      "interface-name": "string",
      "zone": "string",
      "encapsulation": "string"
    }
  ]
}

neighbor_schema:

neighbor-address: string

IPv6 address

mac-address: string

48-bit hardware address H.H.H

interface-name: string

Interface name

zone: string

IPv6 address zone

encapsulation: string srpa, srpb

Encapsulation type only if interface type is SRP

Example
{
  "neighbor-address": "string",
  "mac-address": "string",
  "interface-name": "string",
  "zone": "string",
  "encapsulation": "string"
}
